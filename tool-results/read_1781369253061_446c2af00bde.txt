   210→   210→function formatDate(dateStr: string | null): string {
   211→   211→  if (!dateStr) return '—'
   212→   212→  return new Date(dateStr).toLocaleDateString('en-IN', {
   213→   213→    day: '2-digit',
   214→   214→    month: 'short',
   215→   215→    year: 'numeric',
   216→   216→    hour: '2-digit',
   217→   217→    minute: '2-digit',
   218→   218→  })
   219→   219→}

   221→   221→function formatCurrency(amount: number): string {

   223→   223→    style: 'currency',
   224→   224→    currency: 'INR',
   225→   225→    minimumFractionDigits: 0,
   226→   226→    maximumFractionDigits: 0,

   228→   228→}
