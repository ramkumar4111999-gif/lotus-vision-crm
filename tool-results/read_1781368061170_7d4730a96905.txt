   235→function renderCustomLabel({
   236→  cx,
   237→  cy,
   238→  midAngle,
   239→  innerRadius,
   240→  outerRadius,
   241→  percent,
   242→}: {
   243→  cx: number;
   244→  cy: number;
   245→  midAngle: number;
   246→  innerRadius: number;
   247→  outerRadius: number;
   248→  percent: number;
   249→}) {
   250→  if (percent < 0.06) return null;
   251→  const radius = innerRadius + (outerRadius - innerRadius) * 1.35;
   252→  const x = cx + radius * Math.cos(-midAngle * RADIAN);
   253→  const y = cy + radius * Math.sin(-midAngle * RADIAN);
   254→  return (
   255→    <text x={x} y={y} fill="#374151" textAnchor={x > cx ? "start" : "end"} dominantBaseline="central" className="text-xs">
   256→      {`${(percent * 100).toFixed(0)}%`}
   257→    </text>
   258→  );
   259→}

   342→function useFetch<T>(url: string | null, fallback: T): { data: T; loading: boolean; error: string | null } {
   343→  const [data, setData] = useState<T>(fallback);
   344→  const [loading, setLoading] = useState(true);
   345→  const [error, setError] = useState<string | null>(null);
   346→
   347→  useEffect(() => {
   348→    if (!url) { setLoading(false); return; }
   349→    let cancelled = false;
   350→    setLoading(true);
   351→    setError(null);
   352→    fetch(url)
   353→      .then((r) => {
   354→        if (!r.ok) throw new Error(`HTTP ${r.status}`);
   355→        return r.json();
   356→      })
   357→      .then((d) => { if (!cancelled) setData(d); })
   358→      .catch((e) => { if (!cancelled) setError(e.message); })
   359→      .finally(() => { if (!cancelled) setLoading(false); });
   360→    return () => { cancelled = true; };
   361→  }, [url]);
   362→
   363→  return { data, loading, error };
   364→}

   366→function useReportsData<T>(type: string, fallbackData: T, dateFrom?: string, dateTo?: string) {
   367→  const params = new URLSearchParams({ type });
   368→  if (dateFrom) params.set("from", dateFrom);
   369→  if (dateTo) params.set("to", dateTo);
   370→  return useFetch<T>(`/api/reports?${params}`, fallbackData);
   371→}

   373→function useSalesData(dateFrom?: string, dateTo?: string) {
   374→  const params = new URLSearchParams({ limit: "1000" });
   375→  if (dateFrom) params.set("fromDate", dateFrom);
   376→  if (dateTo) params.set("toDate", dateTo);
   377→  return useFetch<{ sales?: SaleRecord[]; data?: SaleRecord[] }>(`/api/sales?${params}`, { sales: [], data: [] });
   378→}

   430→// ─── PDF Export ───────────────────────────────────────────────────────────────

   432→function usePDFExport() {
   433→  const [exporting, setExporting] = useState(false);
   434→
   435→  const exportPDF = useCallback(async (
   436→    title: string,
   437→    elementId: string,
   438→  ) => {
   439→    setExporting(true);
   440→    try {
   441→      const element = document.getElementById(elementId);
   442→      if (!element) {
   443→        toast.error("Could not find report content to export");
   444→        return;
   445→      }
   446→
   447→      // Use browser print as PDF — works universally without extra libs
   448→      const printWindow = window.open("", "_blank");
   449→      if (!printWindow) {
   450→        toast.error("Please allow popups for PDF export");
   451→        return;
   452→      }
   453→
   454→      const styles = Array.from(document.styleSheets)
   455→        .map((sheet) => {
   456→          try {
   457→            return Array.from(sheet.cssRules).map((rule) => rule.cssText).join("\n");
   458→          } catch {
   459→            return "";
   460→          }
   461→        })
   462→        .join("\n");
   463→
   464→      printWindow.document.write(`
   465→        <!DOCTYPE html>
   466→        <html>
   467→        <head>
   468→          <title>${title} - Lotus Vision Opticals</title>
   469→          <style>
   470→            ${styles}
   471→            body { font-family: -apple-system, BlinkMacSystemFont, "Segoe UI", sans-serif; padding: 20px; color: #111; background: #fff; }
   472→            .print-header { text-align: center; margin-bottom: 24px; border-bottom: 2px solid #059669; padding-bottom: 12px; }
   473→            .print-header h1 { font-size: 20px; margin: 0; color: #059669; }
   474→            .print-header p { font-size: 12px; color: #666; margin: 4px 0 0; }
   475→            table { width: 100%; border-collapse: collapse; font-size: 12px; }
   476→            th, td { border: 1px solid #ddd; padding: 6px 8px; text-align: left; }
   477→            th { background: #f3f4f6; font-weight: 600; }
   478→            tr:nth-child(even) { background: #f9fafb; }
   479→            .text-right { text-align: right; }
   480→            .font-mono { font-family: monospace; }
   481→            .no-print { display: none !important; }
   482→            @media print { body { padding: 0; } }
   483→          </style>
   484→        </head>
   485→        <body>
   486→          <div class="print-header">
   487→            <h1>Lotus Vision Opticals</h1>
   488→            <p>${title} | Generated: ${new Date().toLocaleString("en-IN", { dateStyle: "medium", timeStyle: "short" })}</p>
   489→          </div>
   490→          ${element.innerHTML}
   491→          <script>
   492→            setTimeout(() => { window.print(); window.close(); }, 500);
   493→          </script>
   494→        </body>
   495→        </html>
   496→      `);
   497→      printWindow.document.close();
   498→    } catch {
   499→      toast.error("Failed to generate PDF");
   500→    } finally {
   501→      setExporting(false);
   502→    }
   503→  }, []);
   504→
   505→  return { exporting, exportPDF };
   506→}
