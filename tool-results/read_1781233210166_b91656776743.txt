   990тЖТ   990тЖТfunction KeyboardShortcuts() {
   991тЖТ   991тЖТ  const { focusSearchInput, triggerNewSaleDialog, setSettingsOpen, settingsOpen, setActiveSection } = useCrmStore();
   992тЖТ   992тЖТ
   993тЖТ   993тЖТ  useEffect(() => {
   994тЖТ   994тЖТ    function handleKeyDown(e: KeyboardEvent) {
   995тЖТ   995тЖТ      // Don't trigger shortcuts when typing in inputs/textareas
   996тЖТ   996тЖТ      const tag = (e.target as HTMLElement)?.tagName;
   997тЖТ   997тЖТ      const isInput = tag === 'INPUT' || tag === 'TEXTAREA' || tag === 'SELECT';
   998тЖТ   998тЖТ
   999тЖТ   999тЖТ      // Ctrl+K or Cmd+K: Focus search
  1000тЖТ  1000тЖТ      if ((e.ctrlKey || e.metaKey) && e.key === 'k') {
  1001тЖТ  1001тЖТ        e.preventDefault();
  1002тЖТ  1002тЖТ        focusSearchInput();
  1003тЖТ  1003тЖТ        return;
  1004тЖТ  1004тЖТ      }
  1005тЖТ  1005тЖТ
  1006тЖТ  1006тЖТ      // Ctrl+N: New Sale
  1007тЖТ  1007тЖТ      if (e.ctrlKey && e.key === 'n') {
  1008тЖТ  1008тЖТ        e.preventDefault();
  1009тЖТ  1009тЖТ        triggerNewSaleDialog();
  1010тЖТ  1010тЖТ        return;
  1011тЖТ  1011тЖТ      }
  1012тЖТ  1012тЖТ
  1013тЖТ  1013тЖТ      // Escape: Close settings dialog
  1014тЖТ  1014тЖТ      if (e.key === 'Escape' && settingsOpen) {
  1015тЖТ  1015тЖТ        setSettingsOpen(false);
  1016тЖТ  1016тЖТ        return;
  1017тЖТ  1017тЖТ      }
  1018тЖТ  1018тЖТ
  1019тЖТ  1019тЖТ      // Number keys 1-9: Quick section switch (only when not in input)
  1020тЖТ  1020тЖТ      if (!isInput && !e.ctrlKey && !e.metaKey && !e.altKey) {
  1021тЖТ  1021тЖТ        const num = parseInt(e.key);
  1022тЖТ  1022тЖТ        if (num >= 1 && num <= 9 && NAV_ITEMS[num - 1]) {
  1023тЖТ  1023тЖТ          setActiveSection(NAV_ITEMS[num - 1].key);
  1024тЖТ  1024тЖТ          return;
  1025тЖТ  1025тЖТ        }
  1026тЖТ  1026тЖТ        // 0 = section 10 (Staff)
  1027тЖТ  1027тЖТ        if (e.key === '0' && NAV_ITEMS[9]) {
  1028тЖТ  1028тЖТ          setActiveSection(NAV_ITEMS[9].key);
  1029тЖТ  1029тЖТ          return;
  1030тЖТ  1030тЖТ        }
  1031тЖТ  1031тЖТ      }
  1032тЖТ  1032тЖТ    }
  1033тЖТ  1033тЖТ
  1034тЖТ  1034тЖТ    document.addEventListener('keydown', handleKeyDown);
  1035тЖТ  1035тЖТ    return () => document.removeEventListener('keydown', handleKeyDown);
  1036тЖТ  1036тЖТ  }, [focusSearchInput, triggerNewSaleDialog, setSettingsOpen, settingsOpen, setActiveSection]);
  1037тЖТ  1037тЖТ
  1038тЖТ  1038тЖТ  return null;
  1039тЖТ  1039тЖТ}
