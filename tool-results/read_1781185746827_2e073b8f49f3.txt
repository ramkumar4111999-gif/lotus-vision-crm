   193→// ─── Helpers ─────────────────────────────────────────────────────────────────
   194→
   195→function formatINR(amount: number): string {
   196→  return new Intl.NumberFormat('en-IN', {
   197→    style: 'currency',
   198→    currency: 'INR',
   199→    minimumFractionDigits: 0,
   200→    maximumFractionDigits: 0,
   201→  }).format(amount);
   202→}
   203→
   204→function formatDate(dateStr: string | Date): string {
   205→  try {
   206→    const d = typeof dateStr === 'string' ? parseISO(dateStr) : dateStr;
   207→    return format(d, 'dd MMM yyyy');
   208→  } catch {
   209→    return String(dateStr);
   210→  }
   211→}
   212→
   213→function toDateString(dateStr: string | Date): string {
   214→  try {
   215→    const d = typeof dateStr === 'string' ? new Date(dateStr) : dateStr;
   216→    return format(d, 'yyyy-MM-dd');
   217→  } catch {
   218→    return String(dateStr);
   219→  }
   220→}
   221→
   222→function getTodayStr(): string {
   223→  return format(new Date(), 'yyyy-MM-dd');
   224→}
   225→
   226→function getBudgetKey(): string {
   227→  const now = new Date();
   228→  return `accounting_budget_${now.getFullYear()}_${now.getMonth() + 1}`;
   229→}
