   238→   238→function renderCustomLabel({
   239→   239→  cx,
   240→   240→  cy,
   241→   241→  midAngle,
   242→   242→  innerRadius,
   243→   243→  outerRadius,
   244→   244→  percent,
   245→   245→}: {
   246→   246→  cx: number;
   247→   247→  cy: number;
   248→   248→  midAngle: number;
   249→   249→  innerRadius: number;
   250→   250→  outerRadius: number;
   251→   251→  percent: number;
   252→   252→}) {
   253→   253→  if (percent < 0.06) return null;
   254→   254→  const radius = innerRadius + (outerRadius - innerRadius) * 1.35;
   255→   255→  const x = cx + radius * Math.cos(-midAngle * RADIAN);
   256→   256→  const y = cy + radius * Math.sin(-midAngle * RADIAN);
   257→   257→  return (
   258→   258→    <text x={x} y={y} fill="#374151" textAnchor={x > cx ? "start" : "end"} dominantBaseline="central" className="text-xs">
   259→   259→      {`${(percent * 100).toFixed(0)}%`}
   260→   260→    </text>
   261→   261→  );
   262→   262→}

   353→   353→    },
   354→   354→    placeholderData: fallbackData,
   355→   355→    staleTime: 30_000,
   356→   356→  });

   437→   437→function usePDFExport() {
   438→   438→  const [exporting, setExporting] = useState(false);
   439→   439→
   440→   440→  const exportPDF = useCallback(async (
   441→   441→    title: string,
   442→   442→    elementId: string,
   443→   443→  ) => {
   444→   444→    setExporting(true);
   445→   445→    try {
   446→   446→      const element = document.getElementById(elementId);
   447→   447→      if (!element) {
   448→   448→        toast.error("Could not find report content to export");
   449→   449→        return;
   450→   450→      }
   451→   451→
   452→   452→      // Use browser print as PDF — works universally without extra libs
   453→   453→      const printWindow = window.open("", "_blank");
   454→   454→      if (!printWindow) {
   455→   455→        toast.error("Please allow popups for PDF export");
   456→   456→        return;
   457→   457→      }
   458→   458→
   459→   459→      const styles = Array.from(document.styleSheets)
   460→   460→        .map((sheet) => {
   461→   461→          try {
   462→   462→            return Array.from(sheet.cssRules).map((rule) => rule.cssText).join("\n");
   463→   463→          } catch {
   464→   464→            return "";
   465→   465→          }
   466→   466→        })
   467→   467→        .join("\n");
   468→   468→
   469→   469→      printWindow.document.write(`
   470→   470→        <!DOCTYPE html>
   471→   471→        <html>
   472→   472→        <head>
   473→   473→          <title>${title} - Lotus Vision Opticals</title>
   474→   474→          <style>
   475→   475→            ${styles}
   476→   476→            body { font-family: -apple-system, BlinkMacSystemFont, "Segoe UI", sans-serif; padding: 20px; color: #111; background: #fff; }
   477→   477→            .print-header { text-align: center; margin-bottom: 24px; border-bottom: 2px solid #059669; padding-bottom: 12px; }
   478→   478→            .print-header h1 { font-size: 20px; margin: 0; color: #059669; }
   479→   479→            .print-header p { font-size: 12px; color: #666; margin: 4px 0 0; }
   480→   480→            table { width: 100%; border-collapse: collapse; font-size: 12px; }
   481→   481→            th, td { border: 1px solid #ddd; padding: 6px 8px; text-align: left; }
   482→   482→            th { background: #f3f4f6; font-weight: 600; }
   483→   483→            tr:nth-child(even) { background: #f9fafb; }
   484→   484→            .text-right { text-align: right; }
   485→   485→            .font-mono { font-family: monospace; }
   486→   486→            .no-print { display: none !important; }
   487→   487→            @media print { body { padding: 0; } }
   488→   488→          </style>
   489→   489→        </head>
   490→   490→        <body>
   491→   491→          <div class="print-header">
   492→   492→            <h1>Lotus Vision Opticals</h1>
   493→   493→            <p>${title} | Generated: ${new Date().toLocaleString("en-IN", { dateStyle: "medium", timeStyle: "short" })}</p>
   494→   494→          </div>
   495→   495→          ${element.innerHTML}
   496→   496→          <script>
   497→   497→            setTimeout(() => { window.print(); window.close(); }, 500);
   498→   498→          </script>
   499→   499→        </body>
   500→   500→        </html>
   501→   501→      `);
   502→   502→      printWindow.document.close();
   503→   503→    } catch {
   504→   504→      toast.error("Failed to generate PDF");
   505→   505→    } finally {
   506→   506→      setExporting(false);
   507→   507→    }
   508→   508→  }, []);
   509→   509→
   510→   510→  return { exporting, exportPDF };
   511→   511→}
