   200→function formatINR(amount: number): string {
   201→  return new Intl.NumberFormat('en-IN', {
   202→    style: 'currency',
   203→    currency: 'INR',
   204→    minimumFractionDigits: 0,
   205→    maximumFractionDigits: 0,
   206→  }).format(amount);
   207→}
   208→
   209→function formatDate(dateStr: string | Date): string {
   210→  try {
   211→    const d = typeof dateStr === 'string' ? parseISO(dateStr) : dateStr;
   212→    return format(d, 'dd MMM yyyy');
   213→  } catch {
   214→    return String(dateStr);
   215→  }
   216→}
   217→
   218→function toDateString(dateStr: string | Date): string {
   219→  try {
   220→    const d = typeof dateStr === 'string' ? new Date(dateStr) : dateStr;
   221→    return format(d, 'yyyy-MM-dd');
   222→  } catch {
   223→    return String(dateStr);
   224→  }
   225→}
   226→
   227→function getTodayStr(): string {
   228→  return format(new Date(), 'yyyy-MM-dd');
   229→}
   230→
   231→function getBudgetKey(): string {
   232→  const now = new Date();
   233→  return `accounting_budget_${now.getFullYear()}_${now.getMonth() + 1}`;
   234→}
