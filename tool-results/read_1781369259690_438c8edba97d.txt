   207тЖТ   207тЖТfunction formatTime(timeStr: string | null | undefined): string {
   208тЖТ   208тЖТ  if (!timeStr) return 'тАФ'
   209тЖТ   209тЖТ  try {
   210тЖТ   210тЖТ    const [hours, minutes] = timeStr.split(':').map(Number)
   211тЖТ   211тЖТ    const period = hours >= 12 ? 'PM' : 'AM'
   212тЖТ   212тЖТ    const displayHours = hours % 12 || 12
   213тЖТ   213тЖТ    return `${displayHours}:${minutes.toString().padStart(2, '0')} ${period}`
   214тЖТ   214тЖТ  } catch {
   215тЖТ   215тЖТ    return timeStr
   216тЖТ   216тЖТ  }
   217тЖТ   217тЖТ}
