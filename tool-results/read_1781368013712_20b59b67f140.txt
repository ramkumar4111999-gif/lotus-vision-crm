   176→function formatCurrency(amount: number): string {

   185→function formatDate(dateStr: string | null): string {
   186→  if (!dateStr) return '—'
   187→  return new Date(dateStr).toLocaleDateString('en-IN', {
   188→    day: '2-digit',
   189→    month: 'short',
   190→    year: 'numeric',
   191→  })
   192→}

   214→function getCurrentMonthRange(): { start: Date; end: Date } {
   215→  const now = new Date()
   216→  const start = new Date(now.getFullYear(), now.getMonth(), 1)
   217→  const end = new Date(now.getFullYear(), now.getMonth() + 1, 0, 23, 59, 59, 999)
   218→  return { start, end }
   219→}
   220→
   221→function getElapsedSeconds(isoStr: string): number {
   222→  return Math.max(0, Math.floor((Date.now() - new Date(isoStr).getTime()) / 1000))
   223→}
   224→
   225→function formatElapsed(totalSeconds: number): string {
   226→  const h = Math.floor(totalSeconds / 3600)
   227→  const m = Math.floor((totalSeconds % 3600) / 60)
   228→  const s = totalSeconds % 60
   229→  if (h > 0) return `${h}h ${String(m).padStart(2, '0')}m ${String(s).padStart(2, '0')}s`
   230→  if (m > 0) return `${m}m ${String(s).padStart(2, '0')}s`
   231→  return `${s}s`
   232→}
   233→
   234→function formatHoursDecimal(hours: number): string {
   235→  const h = Math.floor(hours)
   236→  const m = Math.round((hours - h) * 60)
   237→  if (h > 0 && m > 0) return `${h}h ${m}m`
   238→  if (h > 0) return `${h}h`
   239→  return `${m}m`
   240→}

   259→function SalaryManagementSection({ staffList, performanceData }: { staffList: Staff[]; performanceData: StaffPerformance[] }) {
   260→  const [records, setRecords] = useState<SalaryRecord[]>([])
   261→  const [loading, setLoading] = useState(true)
   262→  const [selectedMonth, setSelectedMonth] = useState(() => {
   263→    const now = new Date()

   265→  })
   266→  const [genLoading, setGenLoading] = useState(false)
   267→
   268→  const fetchRecords = useCallback(async () => {
   269→    try {
   270→      const res = await fetch(`/api/staff/salary?month=${selectedMonth}`)
   271→      if (res.ok) {
   272→        const json = await res.json()
   273→        setRecords(json.data ?? [])
   274→      }
   275→    } catch { /* silent */ }
   276→    finally { setLoading(false) }
   277→  }, [selectedMonth])
   278→
   279→  useEffect(() => { fetchRecords() }, [fetchRecords])
   280→
   281→  const handleGenerate = async () => {
   282→    setGenLoading(true)
   283→    try {
   284→      for (const staff of staffList.filter(s => s.isActive)) {
   285→        const perf = performanceData.find(p => p.staffId === staff.id)
   286→        const netPay = staff.salary + (perf?.commissionEarned ?? 0)
   287→        const res = await fetch('/api/staff/salary', {
   288→          method: 'POST',
   289→          headers: { 'Content-Type': 'application/json' },
   290→          body: JSON.stringify({
   291→            staffId: staff.id,
   292→            month: selectedMonth,
   293→            basicSalary: staff.salary,
   294→            commission: Math.round(perf?.commissionEarned ?? 0),
   295→            deductions: 0,
   296→            advance: 0,
   297→            netPay: Math.round(netPay),
   298→          }),
   299→        })
   300→        if (!res.ok) {
   301→          const err = await res.json().catch(() => ({ error: 'Failed' }))
   302→          toast.error(`Failed to generate salary for ${staff.name}: ${err.error}`)
   303→        }
   304→      }
   305→      await fetchRecords()
   306→      toast.success('Salary generation completed')
   307→    } catch (err) {
   308→      toast.error('Network error while generating salaries')
   309→    } finally { setGenLoading(false) }
   310→  }
   311→
   312→  const handleMarkPaid = async (id: string) => {
   313→    try {
   314→      const res = await fetch('/api/staff/salary', {
   315→        method: 'PATCH',
   316→        headers: { 'Content-Type': 'application/json' },
   317→        body: JSON.stringify({ id, status: 'Paid' }),
   318→      })
   319→      if (!res.ok) {
   320→        const err = await res.json().catch(() => ({ error: 'Failed' }))
   321→        toast.error(`Failed to mark as paid: ${err.error}`)
   322→        return
   323→      }
   324→      await fetchRecords()
   325→      toast.success('Salary marked as paid')
   326→    } catch {
   327→      toast.error('Network error while updating salary status')
   328→    }
   329→  }
   330→
   331→  const totalNet = records.reduce((sum, r) => sum + r.netPay, 0)
   332→  const paidCount = records.filter(r => r.status === 'Paid').length
   333→  const pendingCount = records.filter(r => r.status === 'Pending').length
   334→
   335→  return (
   336→    <Card>
   337→      <CardHeader className="flex flex-col gap-4 sm:flex-row sm:items-center sm:justify-between">
   338→        <div className="space-y-1">
   339→          <CardTitle className="flex items-center gap-2 text-xl">
   340→            <Receipt className="size-5 text-primary" />
   341→            Salary Management
   342→          </CardTitle>
   343→          <CardDescription>
   344→            {records.length} record{records.length !== 1 ? 's' : ''} for {new Date(selectedMonth + '-01').toLocaleDateString('en-IN', { month: 'long', year: 'numeric' })}
   345→            {records.length > 0 && ` — Total: ${formatCurrency(totalNet)}`}
   346→          </CardDescription>
   347→        </div>
   348→        <div className="flex items-center gap-2">
   349→          <Input
   350→            type="month"
   351→            value={selectedMonth}
   352→            onChange={(e) => setSelectedMonth(e.target.value)}
   353→            className="w-auto"
   354→          />
   355→          <Button onClick={handleGenerate} disabled={genLoading || staffList.length === 0} variant="outline" className="min-w-[44px] min-h-[44px] touch-manipulation">
   356→            {genLoading ? <span className="size-4 animate-spin inline-block border-2 border-current border-t-transparent rounded-full" /> : <DollarSign className="size-4" />}
   357→            Generate
   358→          </Button>
   359→        </div>
   360→      </CardHeader>
   361→      <CardContent>
   362→        {loading ? (
   363→          <div className="flex items-center justify-center py-8">
   364→            <span className="size-5 animate-spin inline-block border-2 border-current border-t-transparent rounded-full" />
   365→          </div>
   366→        ) : records.length === 0 ? (
   367→          <div className="text-center py-8 text-muted-foreground">
   368→            <Receipt className="size-10 mx-auto mb-2 opacity-30" />
   369→            <p className="text-sm">No salary records for this month.</p>
   370→            <p className="text-xs mt-1">Click &quot;Generate&quot; to create salary records for all active staff.</p>
   371→          </div>
   372→        ) : (
   373→          <div className="space-y-4">
   374→            <div className="flex gap-4 text-sm">
   375→              <Badge variant="outline" className="bg-emerald-50 text-emerald-700 dark:bg-emerald-900/20 dark:text-emerald-400">
   376→                <CheckCircle2 className="size-3 mr-1" /> {paidCount} Paid
   377→              </Badge>
   378→              <Badge variant="outline" className="bg-amber-50 text-amber-700 dark:bg-amber-900/20 dark:text-amber-400">
   379→                <XCircle className="size-3 mr-1" /> {pendingCount} Pending

   381→            </div>
   382→            <div className="overflow-x-auto">
   383→            <Table>
