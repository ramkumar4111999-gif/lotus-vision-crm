   238→function renderCustomLabel({
   239→  cx,
   240→  cy,
   241→  midAngle,
   242→  innerRadius,
   243→  outerRadius,
   244→  percent,
   245→}: {
   246→  cx: number;
   247→  cy: number;
   248→  midAngle: number;
   249→  innerRadius: number;
   250→  outerRadius: number;
   251→  percent: number;
   252→}) {
   253→  if (percent < 0.06) return null;
   254→  const radius = innerRadius + (outerRadius - innerRadius) * 1.35;
   255→  const x = cx + radius * Math.cos(-midAngle * RADIAN);
   256→  const y = cy + radius * Math.sin(-midAngle * RADIAN);
   257→  return (
   258→    <text x={x} y={y} fill="#374151" textAnchor={x > cx ? "start" : "end"} dominantBaseline="central" className="text-xs">
   259→      {`${(percent * 100).toFixed(0)}%`}
   260→    </text>
   261→  );
   262→}

   266→function ChartTooltip({
   267→  active,
   268→  payload,
   269→  label,

   345→function useReportsData<T>(type: string, fallbackData: T, dateFrom?: string, dateTo?: string) {
   346→  return useQuery<T>({
   347→    queryKey: ["reports", type, dateFrom, dateTo],
   348→    queryFn: () => {
   349→      const params = new URLSearchParams({ type });
   350→      if (dateFrom) params.set("from", dateFrom);
   351→      if (dateTo) params.set("to", dateTo);
   352→      return fetch(`/api/reports?${params}`).then((r) => r.json());
   353→    },
   354→    placeholderData: fallbackData,
   355→    staleTime: 30_000,
   356→  });
   357→}

   448→// ─── PDF Export ───────────────────────────────────────────────────────────────
   449→
   450→function usePDFExport() {
   451→  const [exporting, setExporting] = useState(false);
   452→
   453→  const exportPDF = useCallback(async (
   454→    title: string,
   455→    elementId: string,
   456→  ) => {
   457→    setExporting(true);
   458→    try {
   459→      const element = document.getElementById(elementId);
   460→      if (!element) {
   461→        toast.error("Could not find report content to export");
   462→        return;
   463→      }
   464→
   465→      // Use browser print as PDF — works universally without extra libs
   466→      const printWindow = window.open("", "_blank");
   467→      if (!printWindow) {
   468→        toast.error("Please allow popups for PDF export");
   469→        return;
   470→      }
   471→
   472→      const styles = Array.from(document.styleSheets)
   473→        .map((sheet) => {
   474→          try {
   475→            return Array.from(sheet.cssRules).map((rule) => rule.cssText).join("\n");
   476→          } catch {
   477→            return "";
   478→          }
   479→        })
   480→        .join("\n");
   481→
   482→      printWindow.document.write(`
   483→        <!DOCTYPE html>
   484→        <html>
   485→        <head>
   486→          <title>${title} - Lotus Vision Opticals</title>
   487→          <style>
   488→            ${styles}
   489→            body { font-family: -apple-system, BlinkMacSystemFont, "Segoe UI", sans-serif; padding: 20px; color: #111; background: #fff; }
   490→            .print-header { text-align: center; margin-bottom: 24px; border-bottom: 2px solid #059669; padding-bottom: 12px; }
   491→            .print-header h1 { font-size: 20px; margin: 0; color: #059669; }
   492→            .print-header p { font-size: 12px; color: #666; margin: 4px 0 0; }
   493→            table { width: 100%; border-collapse: collapse; font-size: 12px; }
   494→            th, td { border: 1px solid #ddd; padding: 6px 8px; text-align: left; }
   495→            th { background: #f3f4f6; font-weight: 600; }
   496→            tr:nth-child(even) { background: #f9fafb; }
   497→            .text-right { text-align: right; }
   498→            .font-mono { font-family: monospace; }
   499→            .no-print { display: none !important; }
   500→            @media print { body { padding: 0; } }
   501→          </style>
   502→        </head>
   503→        <body>
   504→          <div class="print-header">
   505→            <h1>Lotus Vision Opticals</h1>
   506→            <p>${title} | Generated: ${new Date().toLocaleString("en-IN", { dateStyle: "medium", timeStyle: "short" })}</p>
   507→          </div>
   508→          ${element.innerHTML}
   509→          <script>
   510→            setTimeout(() => { window.print(); window.close(); }, 500);
   511→          </script>
   512→        </body>
   513→        </html>
   514→      `);
   515→      printWindow.document.close();
   516→    } catch {
   517→      toast.error("Failed to generate PDF");
   518→    } finally {
   519→      setExporting(false);
   520→    }
   521→  }, []);
   522→
   523→  return { exporting, exportPDF };
   524→}

   590→    if (raw && "products" in raw && Array.isArray(raw.products)) return raw.products;
   591→    if (raw && "data" in raw && Array.isArray(raw.data)) return raw.data;

   598→    const raw = revenueCompQuery.data;
   599→    if (!raw || !raw.summary) return null;

   602→
   603→  // Product performance data
