   998тЖТ   998тЖТ

  1001тЖТ  1001тЖТ      // Ctrl+K or Cmd+K: Focus search
  1002тЖТ  1002тЖТ      if ((e.ctrlKey || e.metaKey) && e.key === 'k') {
