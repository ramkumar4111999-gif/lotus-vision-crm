   357тЖТ   357тЖТfunction CustomerPurchaseHistory({ customerId }: { customerId: string | null }) {

   359тЖТ   359тЖТ
   360тЖТ   360тЖТ  React.useEffect(() => {

   364тЖТ   364тЖТ    }
   365тЖТ   365тЖТ    let cancelled = false;
   366тЖТ   366тЖТ    (async () => {

   368тЖТ   368тЖТ        const res = await fetch(`/api/sales?customerId=${customerId}&limit=3`);
   369тЖТ   369тЖТ        if (cancelled) return;

   371тЖТ   371тЖТ          const data = await res.json();
   372тЖТ   372тЖТ          const list = Array.isArray(data.sales) ? data.sales : [];
   373тЖТ   373тЖТ          setPurchases(list.map(function(s) {
   374тЖТ   374тЖТ            const r = s as Record<string, unknown>;
   375тЖТ   375тЖТ            return { date: r.createdAt as string, total: r.total as number };
   376тЖТ   376тЖТ          }));
   377тЖТ   377тЖТ        }
   378тЖТ   378тЖТ      } catch {
   379тЖТ   379тЖТ        // silently ignore
   380тЖТ   380тЖТ      }
   381тЖТ   381тЖТ    })();
   382тЖТ   382тЖТ    return () => { cancelled = true; };
   383тЖТ   383тЖТ  }, [customerId]);
   384тЖТ   384тЖТ
   385тЖТ   385тЖТ  if (!customerId || purchases.length === 0) return null;
   386тЖТ   386тЖТ
