   200→   200→function formatINR(amount: number): string {
   201→   201→  return new Intl.NumberFormat('en-IN', {
   202→   202→    style: 'currency',
   203→   203→    currency: 'INR',
   204→   204→    minimumFractionDigits: 0,
   205→   205→    maximumFractionDigits: 0,
   206→   206→  }).format(amount);
   207→   207→}
   208→   208→
   209→   209→function formatDate(dateStr: string | Date): string {
   210→   210→  try {
   211→   211→    const d = typeof dateStr === 'string' ? parseISO(dateStr) : dateStr;
   212→   212→    return format(d, 'dd MMM yyyy');
   213→   213→  } catch {
   214→   214→    return String(dateStr);
   215→   215→  }
   216→   216→}
   217→   217→
   218→   218→function toDateString(dateStr: string | Date): string {
   219→   219→  try {
   220→   220→    const d = typeof dateStr === 'string' ? new Date(dateStr) : dateStr;
   221→   221→    return format(d, 'yyyy-MM-dd');
   222→   222→  } catch {
   223→   223→    return String(dateStr);
   224→   224→  }
   225→   225→}
