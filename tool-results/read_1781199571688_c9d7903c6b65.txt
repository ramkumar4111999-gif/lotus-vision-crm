   249тЖТ   249тЖТfunction formatRelativeTime(dateStr: string): string {
   250тЖТ   250тЖТ  if (!dateStr) return '';
   251тЖТ   251тЖТ  const now = new Date();
   252тЖТ   252тЖТ  const date = new Date(dateStr);
   253тЖТ   253тЖТ  const diffMs = now.getTime() - date.getTime();
   254тЖТ   254тЖТ  const diffMins = Math.floor(diffMs / 60000);
   255тЖТ   255тЖТ  const diffHours = Math.floor(diffMs / 3600000);
   256тЖТ   256тЖТ  const diffDays = Math.floor(diffMs / 86400000);
   257тЖТ   257тЖТ  if (diffMins < 1) return 'just now';
   258тЖТ   258тЖТ  if (diffMins < 60) return `${diffMins}m ago`;
   259тЖТ   259тЖТ  if (diffHours < 24) return `${diffHours}h ago`;
   260тЖТ   260тЖТ  if (diffDays < 7) return `${diffDays}d ago`;
   261тЖТ   261тЖТ  if (diffDays < 30) return `${Math.floor(diffDays / 7)}w ago`;
   262тЖТ   262тЖТ  if (diffDays < 365) return `${Math.floor(diffDays / 30)}mo ago`;
   263тЖТ   263тЖТ  return `${Math.floor(diffDays / 365)}y ago`;
   264тЖТ   264тЖТ}
