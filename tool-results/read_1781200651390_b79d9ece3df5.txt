   143→   143→function formatCurrency(amount: number): string {
   144→   144→  return new Intl.NumberFormat('en-IN', {
   145→   145→    style: 'currency',
   146→   146→    currency: 'INR',
   147→   147→    minimumFractionDigits: 0,
   148→   148→    maximumFractionDigits: 0,
   149→   149→  }).format(amount)
   150→   150→}
   151→   151→
   152→   152→function formatDate(dateStr: string | null): string {
   153→   153→  if (!dateStr) return '—'
   154→   154→  return new Date(dateStr).toLocaleDateString('en-IN', {
   155→   155→    day: '2-digit',
   156→   156→    month: 'short',
   157→   157→    year: 'numeric',
   158→   158→  })
   159→   159→}
   160→   160→
   161→   161→function formatTime(isoStr: string): string {
   162→   162→  return new Date(isoStr).toLocaleTimeString('en-IN', {
   163→   163→    hour: '2-digit',
   164→   164→    minute: '2-digit',
   165→   165→    hour12: true,
   166→   166→  })
   167→   167→}

   174→   174→function getCurrentMonthRange(): { start: Date; end: Date } {
   175→   175→  const now = new Date()
   176→   176→  const start = new Date(now.getFullYear(), now.getMonth(), 1)
   177→   177→  const end = new Date(now.getFullYear(), now.getMonth() + 1, 0, 23, 59, 59, 999)
   178→   178→  return { start, end }
   179→   179→}

   181→   181→function getElapsedSeconds(isoStr: string): number {
   182→   182→  return Math.max(0, Math.floor((Date.now() - new Date(isoStr).getTime()) / 1000))
   183→   183→}
   184→   184→
   185→   185→function formatElapsed(totalSeconds: number): string {
   186→   186→  const h = Math.floor(totalSeconds / 3600)
   187→   187→  const m = Math.floor((totalSeconds % 3600) / 60)
   188→   188→  const s = totalSeconds % 60
   189→   189→  if (h > 0) return `${h}h ${String(m).padStart(2, '0')}m ${String(s).padStart(2, '0')}s`
   190→   190→  if (m > 0) return `${m}m ${String(s).padStart(2, '0')}s`
   191→   191→  return `${s}s`
   192→   192→}

   240→   240→          method: 'POST',

   243→   243→            staffId: staff.id,
