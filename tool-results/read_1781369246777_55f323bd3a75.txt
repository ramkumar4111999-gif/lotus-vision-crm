   177→function formatCurrency(amount: number): string {
   178→  return new Intl.NumberFormat('en-IN', {
   179→    style: 'currency',

   181→    minimumFractionDigits: 0,

   186→function formatDate(dateStr: string | null): string {
   187→  if (!dateStr) return '—'
   188→  return new Date(dateStr).toLocaleDateString('en-IN', {
   189→    day: '2-digit',
   190→    month: 'short',
   191→    year: 'numeric',
   192→  })
   193→}
   194→
   195→function formatDateShort(dateStr: string): string {

   215→function getCurrentMonthRange(): { start: Date; end: Date } {
   216→  const now = new Date()
   217→  const start = new Date(now.getFullYear(), now.getMonth(), 1)
   218→  const end = new Date(now.getFullYear(), now.getMonth() + 1, 0, 23, 59, 59, 999)
   219→  return { start, end }
   220→}
   221→
   222→function getElapsedSeconds(isoStr: string): number {
   223→  return Math.max(0, Math.floor((Date.now() - new Date(isoStr).getTime()) / 1000))
   224→}
   225→
   226→function formatElapsed(totalSeconds: number): string {
   227→  const h = Math.floor(totalSeconds / 3600)
   228→  const m = Math.floor((totalSeconds % 3600) / 60)
   229→  const s = totalSeconds % 60
   230→  if (h > 0) return `${h}h ${String(m).padStart(2, '0')}m ${String(s).padStart(2, '0')}s`
   231→  if (m > 0) return `${m}m ${String(s).padStart(2, '0')}s`
   232→  return `${s}s`
   233→}
   234→
   235→function formatHoursDecimal(hours: number): string {
   236→  const h = Math.floor(hours)
   237→  const m = Math.round((hours - h) * 60)
   238→  if (h > 0 && m > 0) return `${h}h ${m}m`
   239→  if (h > 0) return `${h}h`
   240→  return `${m}m`
   241→}

   288→        const res = await fetch('/api/staff/salary', {
   289→          method: 'POST',
   290→          headers: { 'Content-Type': 'application/json' },
   291→          body: JSON.stringify({
   292→            staffId: staff.id,
   293→            month: selectedMonth,
   294→            basicSalary: staff.salary,

   296→            deductions: 0,
   297→            advance: 0,

   303→          toast.error(`Failed to generate salary for ${staff.name}: ${err.error}`)
