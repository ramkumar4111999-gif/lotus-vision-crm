   178тЖТ   178тЖТfunction formatPower(sph: number | null, cyl: number | null, axis: number | null): string {
   179тЖТ   179тЖТ  const parts: string[] = []
   180тЖТ   180тЖТ  if (sph !== null && sph !== undefined) parts.push(`${sph > 0 ? '+' : ''}${sph}`)
   181тЖТ   181тЖТ  else parts.push('--')
   182тЖТ   182тЖТ  if (cyl !== null && cyl !== undefined && cyl !== 0) parts.push(`${cyl > 0 ? '+' : ''}${cyl}`)
   183тЖТ   183тЖТ  else parts.push('--')
   184тЖТ   184тЖТ  if (axis !== null && axis !== undefined) parts.push(`x${axis}`)
   185тЖТ   185тЖТ  return parts.join(' / ')
   186тЖТ   186тЖТ}
