   196тЖТ   196тЖТfunction formatTime(timeStr: string | null | undefined): string {
   197тЖТ   197тЖТ  if (!timeStr) return 'тАФ'
   198тЖТ   198тЖТ  try {
   199тЖТ   199тЖТ    const [hours, minutes] = timeStr.split(':').map(Number)
   200тЖТ   200тЖТ    const period = hours >= 12 ? 'PM' : 'AM'
   201тЖТ   201тЖТ    const displayHours = hours % 12 || 12
   202тЖТ   202тЖТ    return `${displayHours}:${minutes.toString().padStart(2, '0')} ${period}`
   203тЖТ   203тЖТ  } catch {
   204тЖТ   204тЖТ    return timeStr
   205тЖТ   205тЖТ  }
   206тЖТ   206тЖТ}

   606тЖТ   606тЖТ        // skip
   607тЖТ   607тЖТ      }
