   238→   238→function renderCustomLabel({
   239→   239→  cx,
   240→   240→  cy,
   241→   241→  midAngle,
   242→   242→  innerRadius,
   243→   243→  outerRadius,
   244→   244→  percent,
   245→   245→}: {
   246→   246→  cx: number;
   247→   247→  cy: number;
   248→   248→  midAngle: number;
   249→   249→  innerRadius: number;
   250→   250→  outerRadius: number;
   251→   251→  percent: number;
   252→   252→}) {
   253→   253→  if (percent < 0.06) return null;
   254→   254→  const radius = innerRadius + (outerRadius - innerRadius) * 1.35;
   255→   255→  const x = cx + radius * Math.cos(-midAngle * RADIAN);
   256→   256→  const y = cy + radius * Math.sin(-midAngle * RADIAN);
   257→   257→  return (
   258→   258→    <text x={x} y={y} fill="#374151" textAnchor={x > cx ? "start" : "end"} dominantBaseline="central" className="text-xs">
   259→   259→      {`${(percent * 100).toFixed(0)}%`}
   260→   260→    </text>
   261→   261→  );
   262→   262→}

   345→   345→function useReportsData<T>(type: string, fallbackData: T, dateFrom?: string, dateTo?: string) {
   346→   346→  return useQuery<T>({
   347→   347→    queryKey: ["reports", type, dateFrom, dateTo],
   348→   348→    queryFn: () => {
   349→   349→      const params = new URLSearchParams({ type });
   350→   350→      if (dateFrom) params.set("from", dateFrom);
   351→   351→      if (dateTo) params.set("to", dateTo);
   352→   352→      return fetch(`/api/reports?${params}`).then((r) => r.json());
   353→   353→    },
   354→   354→    placeholderData: fallbackData,
   355→   355→    staleTime: 30_000,
   356→   356→  });
   357→   357→}
