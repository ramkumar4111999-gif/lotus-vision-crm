   236→function renderCustomLabel({
   237→  cx,
   238→  cy,
   239→  midAngle,
   240→  innerRadius,
   241→  outerRadius,
   242→  percent,
   243→}: {
   244→  cx: number;
   245→  cy: number;
   246→  midAngle: number;
   247→  innerRadius: number;
   248→  outerRadius: number;
   249→  percent: number;
   250→}) {
   251→  if (percent < 0.06) return null;
   252→  const radius = innerRadius + (outerRadius - innerRadius) * 1.35;
   253→  const x = cx + radius * Math.cos(-midAngle * RADIAN);
   254→  const y = cy + radius * Math.sin(-midAngle * RADIAN);
   255→  return (
   256→    <text x={x} y={y} fill="#374151" textAnchor={x > cx ? "start" : "end"} dominantBaseline="central" className="text-xs">
   257→      {`${(percent * 100).toFixed(0)}%`}
   258→    </text>
   259→  );
   260→}

   264→function ChartTooltip({
   265→  active,
   266→  payload,
   267→  label,
   268→  valuePrefix = "",

   280→        <div key={i} className="flex items-center gap-2 text-sm">
   281→          <span className="inline-block size-2.5 rounded-full" style={{ backgroundColor: entry.color }} />

   284→            {valuePrefix}{formatNumber(entry.value)}

   343→function useReportsData<T>(type: string, fallbackData: T, dateFrom?: string, dateTo?: string) {
   344→  return useQuery<T>({
   345→    queryKey: ["reports", type, dateFrom, dateTo],
   346→    queryFn: () => {
   347→      const params = new URLSearchParams({ type });
   348→      if (dateFrom) params.set("from", dateFrom);
   349→      if (dateTo) params.set("to", dateTo);
   350→      return fetch(`/api/reports?${params}`).then((r) => r.json());
   351→    },
   352→    placeholderData: fallbackData,
   353→    staleTime: 30_000,
   354→  });
   355→}

   446→// ─── PDF Export ───────────────────────────────────────────────────────────────
   447→
   448→function usePDFExport() {
   449→  const [exporting, setExporting] = useState(false);
   450→
   451→  const exportPDF = useCallback(async (
   452→    title: string,
   453→    elementId: string,
   454→  ) => {
   455→    setExporting(true);
   456→    try {
   457→      const element = document.getElementById(elementId);
   458→      if (!element) {
   459→        toast.error("Could not find report content to export");
   460→        return;
   461→      }
   462→
   463→      // Use browser print as PDF — works universally without extra libs
   464→      const printWindow = window.open("", "_blank");
   465→      if (!printWindow) {
   466→        toast.error("Please allow popups for PDF export");
   467→        return;
   468→      }
   469→
   470→      const styles = Array.from(document.styleSheets)
   471→        .map((sheet) => {
   472→          try {
   473→            return Array.from(sheet.cssRules).map((rule) => rule.cssText).join("\n");
   474→          } catch {
   475→            return "";
   476→          }
   477→        })
   478→        .join("\n");
   479→
   480→      printWindow.document.write(`
   481→        <!DOCTYPE html>
   482→        <html>
   483→        <head>
   484→          <title>${title} - Lotus Vision Opticals</title>
   485→          <style>
   486→            ${styles}
   487→            body { font-family: -apple-system, BlinkMacSystemFont, "Segoe UI", sans-serif; padding: 20px; color: #111; background: #fff; }
   488→            .print-header { text-align: center; margin-bottom: 24px; border-bottom: 2px solid #059669; padding-bottom: 12px; }
   489→            .print-header h1 { font-size: 20px; margin: 0; color: #059669; }
   490→            .print-header p { font-size: 12px; color: #666; margin: 4px 0 0; }
   491→            table { width: 100%; border-collapse: collapse; font-size: 12px; }
   492→            th, td { border: 1px solid #ddd; padding: 6px 8px; text-align: left; }
   493→            th { background: #f3f4f6; font-weight: 600; }
   494→            tr:nth-child(even) { background: #f9fafb; }
   495→            .text-right { text-align: right; }
   496→            .font-mono { font-family: monospace; }
   497→            .no-print { display: none !important; }
   498→            @media print { body { padding: 0; } }
   499→          </style>
   500→        </head>
   501→        <body>
   502→          <div class="print-header">
   503→            <h1>Lotus Vision Opticals</h1>
   504→            <p>${title} | Generated: ${new Date().toLocaleString("en-IN", { dateStyle: "medium", timeStyle: "short" })}</p>
   505→          </div>
   506→          ${element.innerHTML}
   507→          <script>
   508→            setTimeout(() => { window.print(); window.close(); }, 500);
   509→          </script>
   510→        </body>
   511→        </html>
   512→      `);
   513→      printWindow.document.close();
   514→    } catch {
   515→      toast.error("Failed to generate PDF");
   516→    } finally {
   517→      setExporting(false);
   518→    }
   519→  }, []);
   520→
   521→  return { exporting, exportPDF };
   522→}

   584→  }, [salesQuery.data]);

   591→    return [] as Product[];
   592→  }, [productsQuery.data]);
   593→
   594→  // Revenue comparison data

   597→    if (!raw || !raw.summary) return null;

   601→  // Product performance data
