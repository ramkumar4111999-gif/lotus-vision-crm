   196→function formatINR(amount: number): string {
   197→  return new Intl.NumberFormat('en-IN', {
   198→    style: 'currency',
   199→    currency: 'INR',
   200→    minimumFractionDigits: 0,
   201→    maximumFractionDigits: 0,
   202→  }).format(amount);
   203→}
   204→
   205→function formatDate(dateStr: string | Date): string {
   206→  try {
   207→    const d = typeof dateStr === 'string' ? parseISO(dateStr) : dateStr;
   208→    return format(d, 'dd MMM yyyy');
   209→  } catch {
   210→    return String(dateStr);
   211→  }
   212→}
   213→
   214→function toDateString(dateStr: string | Date): string {
   215→  try {
   216→    const d = typeof dateStr === 'string' ? new Date(dateStr) : dateStr;
   217→    return format(d, 'yyyy-MM-dd');
   218→  } catch {
   219→    return String(dateStr);
   220→  }
   221→}
   222→
   223→function getTodayStr(): string {
   224→  return format(new Date(), 'yyyy-MM-dd');
   225→}

   227→function getBudgetKey(): string {
   228→  const now = new Date();

   236→function InvoiceGSTReport() {
   237→  const [sales, setSales] = useState<Array<Record<string, unknown>>>([]);
   238→  const [loading, setLoading] = useState(true);
   239→  const [month, setMonth] = useState(() => {
   240→    const now = new Date();
   241→    return `${now.getFullYear()}-${String(now.getMonth() + 1).padStart(2, '0')}`;
   242→  });
   243→
   244→  useEffect(() => {
   245→    setLoading(true);
   246→    const start = `${month}-01`;
   247→    const [y, m] = month.split('-').map(Number);
   248→    const lastDay = new Date(y, m, 0).getDate();
   249→    const end = `${month}-${String(lastDay).padStart(2, '0')}`;
   250→    fetch(`/api/sales?fromDate=${start}&toDate=${end}&limit=500`)
   251→      .then(r => r.ok ? r.json() : [])
   252→      .then(json => {
   253→        const data = json.sales ?? json.data ?? json ?? [];
   254→        setSales(Array.isArray(data) ? data : []);
   255→      })
   256→      .catch(() => {})
   257→      .finally(() => setLoading(false));
   258→  }, [month]);
   259→
   260→  const gstSales = sales.filter((s: Record<string, unknown>) => ((s.cgst as number) || 0) + ((s.sgst as number) || 0) + ((s.igst as number) || 0) > 0);
   261→  const totals = gstSales.reduce((acc: Record<string, number>, s: Record<string, unknown>) => {
   262→    acc.subtotal += (s.subtotal as number) || 0;
   263→    acc.cgst += (s.cgst as number) || 0;
   264→    acc.sgst += (s.sgst as number) || 0;
   265→    acc.igst += (s.igst as number) || 0;
   266→    acc.total += (s.totalAmount as number) || 0;
   267→    return acc;
   268→  }, { subtotal: 0, cgst: 0, sgst: 0, igst: 0, total: 0 });
   269→
   270→  if (loading) return <div className="flex items-center justify-center py-8"><span className="size-5 animate-spin inline-block border-2 border-current border-t-transparent rounded-full" /></div>;
   271→
   272→  return (
   273→    <div className="space-y-4">
   274→      <div className="flex items-center gap-2">
   275→        <Input type="month" value={month} onChange={e => setMonth(e.target.value)} className="w-auto" />
   276→        <p className="text-sm text-muted-foreground">{gstSales.length} invoices with GST</p>
   277→      </div>
   278→      {gstSales.length === 0 ? (
   279→        <p className="text-sm text-muted-foreground text-center py-8">No GST invoices for this month.</p>
   280→      ) : (
   281→        <>
   282→          <Table>
   283→            <TableHeader>
   284→              <TableRow>
   285→                <TableHead>Invoice</TableHead>
   286→                <TableHead>Customer</TableHead>
   287→                <TableHead className="text-right">Subtotal</TableHead>
   288→                <TableHead className="text-right">CGST</TableHead>
   289→                <TableHead className="text-right">SGST</TableHead>
   290→                <TableHead className="text-right">IGST</TableHead>
   291→                <TableHead className="text-right">Total</TableHead>
   292→              </TableRow>
   293→            </TableHeader>
   294→            <TableBody>
   295→              {gstSales.map((s: Record<string, unknown>) => (
   296→                <TableRow key={s.id}>
   297→                  <TableCell className="font-mono text-xs">{s.invoiceNo as string}</TableCell>
   298→                  <TableCell className="text-sm">{(s.customerName as string) || 'Walk-in'}</TableCell>
   299→                  <TableCell className="text-right font-mono text-sm">{formatCurrency((s.subtotal as number) || 0)}</TableCell>
   300→                  <TableCell className="text-right font-mono text-sm text-blue-600 dark:text-blue-400">{formatCurrency((s.cgst as number) || 0)}</TableCell>
   301→                  <TableCell className="text-right font-mono text-sm text-blue-600 dark:text-blue-400">{formatCurrency((s.sgst as number) || 0)}</TableCell>
   302→                  <TableCell className="text-right font-mono text-sm text-purple-600 dark:text-purple-400">{formatCurrency((s.igst as number) || 0)}</TableCell>
   303→                  <TableCell className="text-right font-mono text-sm font-semibold">{formatCurrency((s.totalAmount as number) || 0)}</TableCell>
   304→                </TableRow>
   305→              ))}
   306→              <TableRow className="bg-muted/50 font-semibold">
   307→                <TableCell colSpan={2}>Total</TableCell>
   308→                <TableCell className="text-right font-mono text-sm">{formatCurrency(totals.subtotal)}</TableCell>
   309→                <TableCell className="text-right font-mono text-sm">{formatCurrency(totals.cgst)}</TableCell>
   310→                <TableCell className="text-right font-mono text-sm">{formatCurrency(totals.sgst)}</TableCell>
   311→                <TableCell className="text-right font-mono text-sm">{formatCurrency(totals.igst)}</TableCell>

   313→              </TableRow>
   314→            </TableBody>
   315→          </Table>
   316→          <div className="grid grid-cols-2 sm:grid-cols-4 gap-3">
   317→            <div className="rounded-lg border p-3 text-center">
   318→              <p className="text-xs text-muted-foreground">CGST</p>
   319→              <p className="font-mono font-semibold">{formatCurrency(totals.cgst)}</p>
   320→            </div>
   321→            <div className="rounded-lg border p-3 text-center">
   322→              <p className="text-xs text-muted-foreground">SGST</p>
   323→              <p className="font-mono font-semibold">{formatCurrency(totals.sgst)}</p>
   324→            </div>
   325→            <div className="rounded-lg border p-3 text-center">
   326→              <p className="text-xs text-muted-foreground">IGST</p>
   327→              <p className="font-mono font-semibold">{formatCurrency(totals.igst)}</p>

   332→            </div>
   333→          </div>
   334→        </>
   335→      )}
   336→    </div>
   337→  );
   338→}
