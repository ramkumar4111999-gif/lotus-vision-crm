   283тЖТ   283тЖТfunction formatRelativeTime(dateStr: string): string {
   284тЖТ   284тЖТ  if (!dateStr) return '';
   285тЖТ   285тЖТ  const now = new Date();
   286тЖТ   286тЖТ  const date = new Date(dateStr);
   287тЖТ   287тЖТ  const diffMs = now.getTime() - date.getTime();
   288тЖТ   288тЖТ  const diffMins = Math.floor(diffMs / 60000);
   289тЖТ   289тЖТ  const diffHours = Math.floor(diffMs / 3600000);
   290тЖТ   290тЖТ  const diffDays = Math.floor(diffMs / 86400000);
   291тЖТ   291тЖТ  if (diffMins < 1) return 'just now';
   292тЖТ   292тЖТ  if (diffMins < 60) return `${diffMins}m ago`;
   293тЖТ   293тЖТ  if (diffHours < 24) return `${diffHours}h ago`;
   294тЖТ   294тЖТ  if (diffDays < 7) return `${diffDays}d ago`;
   295тЖТ   295тЖТ  if (diffDays < 30) return `${Math.floor(diffDays / 7)}w ago`;
   296тЖТ   296тЖТ  if (diffDays < 365) return `${Math.floor(diffDays / 30)}mo ago`;
   297тЖТ   297тЖТ  return `${Math.floor(diffDays / 365)}y ago`;
   298тЖТ   298тЖТ}
