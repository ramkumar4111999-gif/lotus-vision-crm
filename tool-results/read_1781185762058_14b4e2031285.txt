   188тЖТ   188тЖТfunction formatTime(timeStr: string | null | undefined): string {
   189тЖТ   189тЖТ  if (!timeStr) return 'тАФ'
   190тЖТ   190тЖТ  try {
   191тЖТ   191тЖТ    const [hours, minutes] = timeStr.split(':').map(Number)
   192тЖТ   192тЖТ    const period = hours >= 12 ? 'PM' : 'AM'
   193тЖТ   193тЖТ    const displayHours = hours % 12 || 12
   194тЖТ   194тЖТ    return `${displayHours}:${minutes.toString().padStart(2, '0')} ${period}`
   195тЖТ   195тЖТ  } catch {
   196тЖТ   196тЖТ    return timeStr
   197тЖТ   197тЖТ  }
   198тЖТ   198тЖТ}
