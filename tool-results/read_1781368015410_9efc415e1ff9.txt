   250тЖТ   250тЖТfunction formatRelativeTime(dateStr: string): string {
   251тЖТ   251тЖТ  if (!dateStr) return '';
   252тЖТ   252тЖТ  const now = new Date();
   253тЖТ   253тЖТ  const date = new Date(dateStr);
   254тЖТ   254тЖТ  const diffMs = now.getTime() - date.getTime();
   255тЖТ   255тЖТ  const diffMins = Math.floor(diffMs / 60000);
   256тЖТ   256тЖТ  const diffHours = Math.floor(diffMs / 3600000);
   257тЖТ   257тЖТ  const diffDays = Math.floor(diffMs / 86400000);
   258тЖТ   258тЖТ  if (diffMins < 1) return 'just now';
   259тЖТ   259тЖТ  if (diffMins < 60) return `${diffMins}m ago`;
   260тЖТ   260тЖТ  if (diffHours < 24) return `${diffHours}h ago`;
   261тЖТ   261тЖТ  if (diffDays < 7) return `${diffDays}d ago`;
   262тЖТ   262тЖТ  if (diffDays < 30) return `${Math.floor(diffDays / 7)}w ago`;
   263тЖТ   263тЖТ  if (diffDays < 365) return `${Math.floor(diffDays / 30)}mo ago`;
   264тЖТ   264тЖТ  return `${Math.floor(diffDays / 365)}y ago`;
   265тЖТ   265тЖТ}
