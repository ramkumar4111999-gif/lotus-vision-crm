   237→function renderCustomLabel({
   238→  cx,
   239→  cy,
   240→  midAngle,
   241→  innerRadius,
   242→  outerRadius,
   243→  percent,
   244→}: {
   245→  cx: number;
   246→  cy: number;
   247→  midAngle: number;
   248→  innerRadius: number;
   249→  outerRadius: number;
   250→  percent: number;
   251→}) {
   252→  if (percent < 0.06) return null;
   253→  const radius = innerRadius + (outerRadius - innerRadius) * 1.35;
   254→  const x = cx + radius * Math.cos(-midAngle * RADIAN);
   255→  const y = cy + radius * Math.sin(-midAngle * RADIAN);
   256→  return (
   257→    <text x={x} y={y} fill="#374151" textAnchor={x > cx ? "start" : "end"} dominantBaseline="central" className="text-xs">
   258→      {`${(percent * 100).toFixed(0)}%`}
   259→    </text>
   260→  );
   261→}

   265→function ChartTooltip({
   266→  active,
   267→  payload,
   268→  label,

   344→function useReportsData<T>(type: string, fallbackData: T, dateFrom?: string, dateTo?: string) {
   345→  return useQuery<T>({
   346→    queryKey: ["reports", type, dateFrom, dateTo],
   347→    queryFn: () => {
   348→      const params = new URLSearchParams({ type });
   349→      if (dateFrom) params.set("from", dateFrom);
   350→      if (dateTo) params.set("to", dateTo);
   351→      return fetch(`/api/reports?${params}`).then((r) => r.json());
   352→    },
   353→    placeholderData: fallbackData,
   354→    staleTime: 30_000,
   355→  });
   356→}

   447→// ─── PDF Export ───────────────────────────────────────────────────────────────
   448→
   449→function usePDFExport() {
   450→  const [exporting, setExporting] = useState(false);
   451→
   452→  const exportPDF = useCallback(async (
   453→    title: string,
   454→    elementId: string,
   455→  ) => {
   456→    setExporting(true);
   457→    try {
   458→      const element = document.getElementById(elementId);
   459→      if (!element) {
   460→        toast.error("Could not find report content to export");
   461→        return;
   462→      }
   463→
   464→      // Use browser print as PDF — works universally without extra libs
   465→      const printWindow = window.open("", "_blank");
   466→      if (!printWindow) {
   467→        toast.error("Please allow popups for PDF export");
   468→        return;
   469→      }
   470→
   471→      const styles = Array.from(document.styleSheets)
   472→        .map((sheet) => {
   473→          try {
   474→            return Array.from(sheet.cssRules).map((rule) => rule.cssText).join("\n");
   475→          } catch {
   476→            return "";
   477→          }
   478→        })
   479→        .join("\n");
   480→
   481→      printWindow.document.write(`
   482→        <!DOCTYPE html>
   483→        <html>
   484→        <head>
   485→          <title>${title} - Lotus Vision Opticals</title>
   486→          <style>
   487→            ${styles}
   488→            body { font-family: -apple-system, BlinkMacSystemFont, "Segoe UI", sans-serif; padding: 20px; color: #111; background: #fff; }
   489→            .print-header { text-align: center; margin-bottom: 24px; border-bottom: 2px solid #059669; padding-bottom: 12px; }
   490→            .print-header h1 { font-size: 20px; margin: 0; color: #059669; }
   491→            .print-header p { font-size: 12px; color: #666; margin: 4px 0 0; }
   492→            table { width: 100%; border-collapse: collapse; font-size: 12px; }
   493→            th, td { border: 1px solid #ddd; padding: 6px 8px; text-align: left; }
   494→            th { background: #f3f4f6; font-weight: 600; }
   495→            tr:nth-child(even) { background: #f9fafb; }
   496→            .text-right { text-align: right; }
   497→            .font-mono { font-family: monospace; }
   498→            .no-print { display: none !important; }
   499→            @media print { body { padding: 0; } }
   500→          </style>
   501→        </head>
   502→        <body>
   503→          <div class="print-header">
   504→            <h1>Lotus Vision Opticals</h1>
   505→            <p>${title} | Generated: ${new Date().toLocaleString("en-IN", { dateStyle: "medium", timeStyle: "short" })}</p>
   506→          </div>
   507→          ${element.innerHTML}
   508→          <script>
   509→            setTimeout(() => { window.print(); window.close(); }, 500);
   510→          </script>
   511→        </body>
   512→        </html>
   513→      `);
   514→      printWindow.document.close();
   515→    } catch {
   516→      toast.error("Failed to generate PDF");
   517→    } finally {
   518→      setExporting(false);
   519→    }
   520→  }, []);
   521→
   522→  return { exporting, exportPDF };
   523→}
