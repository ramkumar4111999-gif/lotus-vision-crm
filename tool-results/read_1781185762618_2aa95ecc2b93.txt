   195→   195→function formatINR(amount: number): string {
   196→   196→  return new Intl.NumberFormat('en-IN', {
   197→   197→    style: 'currency',
   198→   198→    currency: 'INR',
   199→   199→    minimumFractionDigits: 0,
   200→   200→    maximumFractionDigits: 0,
   201→   201→  }).format(amount);
   202→   202→}
   203→   203→
   204→   204→function formatDate(dateStr: string | Date): string {
   205→   205→  try {
   206→   206→    const d = typeof dateStr === 'string' ? parseISO(dateStr) : dateStr;
   207→   207→    return format(d, 'dd MMM yyyy');
   208→   208→  } catch {
   209→   209→    return String(dateStr);
   210→   210→  }
   211→   211→}
   212→   212→
   213→   213→function toDateString(dateStr: string | Date): string {
   214→   214→  try {
   215→   215→    const d = typeof dateStr === 'string' ? new Date(dateStr) : dateStr;
   216→   216→    return format(d, 'yyyy-MM-dd');
   217→   217→  } catch {
   218→   218→    return String(dateStr);
   219→   219→  }
   220→   220→}
   221→   221→
   222→   222→function getTodayStr(): string {
   223→   223→  return format(new Date(), 'yyyy-MM-dd');
   224→   224→}

   226→   226→function getBudgetKey(): string {
   227→   227→  const now = new Date();

   229→   229→}
