   186→   186→function formatDate(dateStr: string | null): string {
   187→   187→  if (!dateStr) return '—'
   188→   188→  return new Date(dateStr).toLocaleDateString('en-IN', {
   189→   189→    day: '2-digit',
   190→   190→    month: 'short',
   191→   191→    year: 'numeric',
   192→   192→  })

   215→   215→function getCurrentMonthRange(): { start: Date; end: Date } {
   216→   216→  const now = new Date()

   222→   222→function getElapsedSeconds(isoStr: string): number {
   223→   223→  return Math.max(0, Math.floor((Date.now() - new Date(isoStr).getTime()) / 1000))
   224→   224→}
   225→   225→
   226→   226→function formatElapsed(totalSeconds: number): string {
   227→   227→  const h = Math.floor(totalSeconds / 3600)
   228→   228→  const m = Math.floor((totalSeconds % 3600) / 60)
   229→   229→  const s = totalSeconds % 60
   230→   230→  if (h > 0) return `${h}h ${String(m).padStart(2, '0')}m ${String(s).padStart(2, '0')}s`
   231→   231→  if (m > 0) return `${m}m ${String(s).padStart(2, '0')}s`
   232→   232→  return `${s}s`
   233→   233→}
   234→   234→
   235→   235→function formatHoursDecimal(hours: number): string {
   236→   236→  const h = Math.floor(hours)
   237→   237→  const m = Math.round((hours - h) * 60)
   238→   238→  if (h > 0 && m > 0) return `${h}h ${m}m`
   239→   239→  if (h > 0) return `${h}h`
   240→   240→  return `${m}m`
   241→   241→}

   260→   260→function SalaryManagementSection({ staffList, performanceData }: { staffList: Staff[]; performanceData: StaffPerformance[] }) {
   261→   261→  const [records, setRecords] = useState<SalaryRecord[]>([])
   262→   262→  const [loading, setLoading] = useState(true)
   263→   263→  const [selectedMonth, setSelectedMonth] = useState(() => {
   264→   264→    const now = new Date()
   265→   265→    return `${now.getFullYear()}-${String(now.getMonth() + 1).padStart(2, '0')}`
   266→   266→  })
   267→   267→  const [genLoading, setGenLoading] = useState(false)
   268→   268→
   269→   269→  const fetchRecords = useCallback(async () => {
   270→   270→    try {
   271→   271→      const res = await fetch(`/api/staff/salary?month=${selectedMonth}`)
   272→   272→      if (res.ok) {
   273→   273→        const json = await res.json()
   274→   274→        setRecords(json.data ?? [])
   275→   275→      }
   276→   276→    } catch { /* silent */ }
   277→   277→    finally { setLoading(false) }
   278→   278→  }, [selectedMonth])
   279→   279→
   280→   280→  useEffect(() => { fetchRecords() }, [fetchRecords])
   281→   281→
   282→   282→  const handleGenerate = async () => {
   283→   283→    setGenLoading(true)
   284→   284→    try {
   285→   285→      for (const staff of staffList.filter(s => s.isActive)) {
   286→   286→        const perf = performanceData.find(p => p.staffId === staff.id)
   287→   287→        const netPay = staff.salary + (perf?.commissionEarned ?? 0)
   288→   288→        const res = await fetch('/api/staff/salary', {
   289→   289→          method: 'POST',
   290→   290→          headers: { 'Content-Type': 'application/json' },
   291→   291→          body: JSON.stringify({
   292→   292→            staffId: staff.id,
   293→   293→            month: selectedMonth,
   294→   294→            basicSalary: staff.salary,
   295→   295→            commission: Math.round(perf?.commissionEarned ?? 0),
   296→   296→            deductions: 0,
   297→   297→            advance: 0,
   298→   298→            netPay: Math.round(netPay),
   299→   299→          }),
   300→   300→        })
   301→   301→        if (!res.ok) {
   302→   302→          const err = await res.json().catch(() => ({ error: 'Failed' }))
   303→   303→          toast.error(`Failed to generate salary for ${staff.name}: ${err.error}`)
   304→   304→        }
   305→   305→      }
   306→   306→      await fetchRecords()
   307→   307→      toast.success('Salary generation completed')
   308→   308→    } catch (err) {
   309→   309→      toast.error('Network error while generating salaries')
   310→   310→    } finally { setGenLoading(false) }
   311→   311→  }
   312→   312→
   313→   313→  const handleMarkPaid = async (id: string) => {
   314→   314→    try {
   315→   315→      const res = await fetch('/api/staff/salary', {
   316→   316→        method: 'PATCH',
   317→   317→        headers: { 'Content-Type': 'application/json' },
   318→   318→        body: JSON.stringify({ id, status: 'Paid' }),
   319→   319→      })
   320→   320→      if (!res.ok) {
   321→   321→        const err = await res.json().catch(() => ({ error: 'Failed' }))
   322→   322→        toast.error(`Failed to mark as paid: ${err.error}`)
   323→   323→        return
   324→   324→      }
   325→   325→      await fetchRecords()
   326→   326→      toast.success('Salary marked as paid')
   327→   327→    } catch {
   328→   328→      toast.error('Network error while updating salary status')
   329→   329→    }
   330→   330→  }
   331→   331→
   332→   332→  const totalNet = records.reduce((sum, r) => sum + r.netPay, 0)
   333→   333→  const paidCount = records.filter(r => r.status === 'Paid').length
   334→   334→  const pendingCount = records.filter(r => r.status === 'Pending').length
   335→   335→
   336→   336→  return (
   337→   337→    <Card>
   338→   338→      <CardHeader className="flex flex-col gap-4 sm:flex-row sm:items-center sm:justify-between">
   339→   339→        <div className="space-y-1">
   340→   340→          <CardTitle className="flex items-center gap-2 text-xl">
   341→   341→            <Receipt className="size-5 text-primary" />
   342→   342→            Salary Management
   343→   343→          </CardTitle>
   344→   344→          <CardDescription>
   345→   345→            {records.length} record{records.length !== 1 ? 's' : ''} for {new Date(selectedMonth + '-01').toLocaleDateString('en-IN', { month: 'long', year: 'numeric' })}
   346→   346→            {records.length > 0 && ` — Total: ${formatCurrency(totalNet)}`}
   347→   347→          </CardDescription>
   348→   348→        </div>
   349→   349→        <div className="flex items-center gap-2">
   350→   350→          <Input
   351→   351→            type="month"

   353→   353→            onChange={(e) => setSelectedMonth(e.target.value)}
   354→   354→            className="w-auto"

   356→   356→          <Button onClick={handleGenerate} disabled={genLoading || staffList.length === 0} variant="outline" className="min-w-[44px] min-h-[44px] touch-manipulation">
   357→   357→            {genLoading ? <span className="size-4 animate-spin inline-block border-2 border-current border-t-transparent rounded-full" /> : <DollarSign className="size-4" />}

   377→   377→                <CheckCircle2 className="size-3 mr-1" /> {paidCount} Paid
   378→   378→              </Badge>
