   466тЖТ   466тЖТfunction CustomerPurchaseHistory({ customerId }: { customerId: string | null }) {
   467тЖТ   467тЖТ  const [purchases, setPurchases] = React.useState<{ date: string; total: number }[]>([]);
   468тЖТ   468тЖТ
   469тЖТ   469тЖТ  React.useEffect(() => {
   470тЖТ   470тЖТ    if (!customerId) {

   473тЖТ   473тЖТ    }

   475тЖТ   475тЖТ    (async () => {
   476тЖТ   476тЖТ      try {

   485тЖТ   485тЖТ          }));

   488тЖТ   488тЖТ        // silently ignore
